<script type="text/javascript" src="http://code.jquery.com/jquery-latest.pack.js"></script>
<script>

	$(document).ready(function() {	

	//select all the a tag with name equal to modal
	$('a[href=#lb_dialog]').click(function(e) {
		//Cancel the link behavior
		e.preventDefault();
		
		//Get the A tag
		var id = $(this).attr('href');
	
		//Get the screen height and width
		var maskHeight = $(document).height();
		var maskWidth = $(window).width();
	
		//Set heigth and width to mask to fill up the whole screen
		$('#mask').css({'width':maskWidth,'height':maskHeight});
		
		//transition effect		
		$('#mask').fadeIn(1000);	
		$('#mask').fadeTo("slow",0.8);	
	
		//Get the window height and width
		var winH = $(window).height();
		var winW = $(window).width();
              
		//Set the popup window to center
		$(id).css('top',  winH/2-$(id).height()/2);
		$(id).css('left', winW/2-$(id).width()/2);
	
		//transition effect
		$(id).fadeIn(2000); 
	});
	
	//if mask is clicked
	$('#mask').click(function () {
		$('#mask').hide();
		$('.lb_window').hide();
	});			

	//if the window is clicked	
	$('.lb_window').click(function () {
		$('.lb_window').hide();
		$('#mask').hide();
	});			

});

</script>
<div id="lb_dialog" class="lb_window">
<div style="position:absolute;top:95%;left:85%;color:white;">
        <font size="2" >Photo by Emmanuel Andre</font>
</div>
</div>
